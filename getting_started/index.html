    Quick Start

    The JellyFAAS SDK simplifies working with the JellyFAAS API. It provides a Client class which handles authentication and functions to simplify creating HTTP requests to the API.

    API Credits

    The cloud functions used in the examples below will not use your credits. Click here to get an api key.

    Below are some quick start examples you can plug and play into.

    main.py

    import jellyfaas
import os
api_key = os.getenv("JELLYFAAS_API_KEY")

# Create client config instance
config = jellyfaas.ClientConfig(api_key)

# Create Client instance, lookup cloud function, set the request params/body, and invoke.
client, response = jellyfaas.Client(config).lookup_function("helloworld").set_request({"name": "arthur"}, {"order": ["ham", "jam", "spam"]}).invoke()

print(response)

    More Info 

    import jellyfaas

# It is recomended to set up your API key as an environment variable.
import os
api_key = os.getenv("JELLYFAAS_API_KEY")

# Initialize the client config
config = jellyfaas.ClientConfig(api_key=api_key, debug=True)

# Initialize the client
client = jellyfaas.Client(config)

# Lookup a function
client.lookup_function("helloworld")

# Set the request query params and JSON body (via a python dictionary)
client.set_request(
    {"name": "tim"},                # Set query parameters
    {"order": ["ham", "jam", "spam"]}) # Set request body

# Finally, call the cloud function
_, response = client.invoke()

print(response)
# Output: "Hello, Tim. You ordered 'ham', 'jam', and 'spam'."

    Warning

    You must add the respective input and output classes provided on the website, if there are any, in order to process JSON requests and responses. These are listed below in the example.

    Main.java

    package main;

import com.fasterxml.jackson.core.JsonProcessingException;
import testSchemas.IbanOutputTest;
import testSchemas.IbanTest;
import v1.jellyfaas.exceptions.JellyFaasException;
import v1.jellyfaas.exceptions.JellyFaasFormatException;
import v1.jellyfaas.exceptions.JellyFaasHttpException;

public class Main {
    public static void main(String[] args) throws JellyFaasException, JellyFaasHttpException, IllegalAccessException, JsonProcessingException, JellyFaasFormatException {
        // setting the config object
        String apiKey = System.getenv("API_KEY");
        Config config = new Config();
        config.setApiKey(apiKey);

        //creating the client object
        Client client = new Client(config);
        IbanTest ibanTest = new IbanTest();
        ibanTest.addIban("FR1420041010050500013M02606");

        // invoking the object and putting the output into a new object
        IbanOutputTest output = (IbanOutputTest) client.lookUp("ibanvalidation").setRequest(null, ibanTest).setResponse(IbanOutputTest.class).invoke();
    }
}

    package main;
import v1.jellyfaas.JellyFaasFunctionSchema;

import java.util.ArrayList;
import java.util.List;

public class IbanOutputTest extends JellyFaasFunctionSchema {
    private String spanId;
    private Data data;

    // Getter and Setter for spanId
    public String getSpanId() {
        return spanId;
    }

    public void setSpanId(String spanId) {
        this.spanId = spanId;
    }

    // Getter and Setter for Data
    public Data getData() {
        return data;
    }

    public void setData(Data data) {
        this.data = data;
    }

    // Inner class Data
    public static class Data {
        private List<String> valid = new ArrayList<>();
        private List<String> invalid = new ArrayList<>();
        private int validCount = 0;
        private int invalidCount = 0;
        private int total = 0;

        // Getters and Setters for valid
        public List<String> getValid() {
            return valid;
        }

        public void setValid(List<String> valid) {
            this.valid = valid;
        }

        // Getters and Setters for invalid
        public List<String> getInvalid() {
            return invalid;
        }

        public void setInvalid(List<String> invalid) {
            this.invalid = invalid;
        }

        // Getters and Setters for validCount
        public int getValidCount() {
            return validCount;
        }

        public void setValidCount(int validCount) {
            this.validCount = validCount;
        }

        // Getters and Setters for invalidCount
        public int getInvalidCount() {
            return invalidCount;
        }

        public void setInvalidCount(int invalidCount) {
            this.invalidCount = invalidCount;
        }

        // Getters and Setters for total
        public int getTotal() {
            return total;
        }

        public void setTotal(int total) {
            this.total = total;
        }
    }
}
